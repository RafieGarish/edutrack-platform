import 'dart:ui';

import 'package:flutter/gestures.dart';
import 'package:flutter/material.dart';
import 'package:flutter_floating_bottom_bar/flutter_floating_bottom_bar.dart';
// Add the import for InfiniteListPage from the file where it's defined
import 'infinite_list_page.dart';

class HomePage extends StatefulWidget {
  const HomePage({super.key});

  @override
  State<HomePage> createState() => _HomePageState();
}

class _HomePageState extends State<HomePage> {
  int _selectedIndex = 0;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      extendBody: true,
      body: Stack(
        children: [
          Image.network('https://external-preview.redd.it/-EjNRzTJvXVBgeMNG2o9y6n897PPQmIqS8QrWhKx2Rc.jpg?width=640&crop=smart&auto=webp&s=69ba302e10019fe50037a747e7f6343fbab47376', fit: BoxFit.cover,), 
          Center(
            child: Text(
              "Index: $_selectedIndex",
              style: const TextStyle(
                  fontSize: 30, color: Colors.white, fontWeight: FontWeight.bold),
            ),
          ),
        ],
      ),
      bottomNavigationBar: GlassFloatingNavBar(
        currentIndex: _selectedIndex,
        onTap: (index) {
          setState(() {
            _selectedIndex = index;
          });
        },
      ),
    );
  }
}

class GlassFloatingNavBar extends StatelessWidget {
  final int currentIndex;
  final Function(int) onTap;

  const GlassFloatingNavBar({
    super.key,
    required this.currentIndex,
    required this.onTap,
  });

  final List<IconData> icons = const [
    Icons.home_rounded,
    Icons.search_rounded,
    Icons.favorite_rounded,
    Icons.person_rounded,
  ];

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.fromLTRB(20, 0, 20, 30),
      child: ClipRRect(
        borderRadius: BorderRadius.circular(30),
        child: BackdropFilter(
          filter: ImageFilter.blur(sigmaX: 15, sigmaY: 15),
          child: Container(
            height: 70, // Fixed height for the bar
            decoration: BoxDecoration(
              gradient: LinearGradient(
                begin: Alignment.topLeft,
                end: Alignment.bottomRight,
                colors: [
                  Colors.white.withOpacity(0.2),
                  Colors.white.withOpacity(0.05),
                ],
              ),
              borderRadius: BorderRadius.circular(30),
              border: Border.all(
                width: 1.5,
                color: Colors.white.withOpacity(0.2),
              ),
              boxShadow: [
                BoxShadow(
                  color: Colors.black.withOpacity(0.1),
                  blurRadius: 20,
                  spreadRadius: 5,
                )
              ],
            ),
            // LayoutBuilder is needed to measure the width for the slide calculation
            child: LayoutBuilder(
              builder: (context, constraints) {
                // Calculate how wide one item slot is
                double itemWidth = constraints.maxWidth / icons.length;

                return Stack(
                  children: [
                    // -------------------------------------------
                    // LAYER 1: The Sliding Indicator (Background)
                    // -------------------------------------------
                    AnimatedPositioned(
                      duration: const Duration(milliseconds: 300),
                      curve: Curves.easeInOut, // Bouncy effect
                      top: 10,
                      bottom: 10,
                      // Calculate left position: Slot Width * Index
                      left: itemWidth * currentIndex, 
                      width: itemWidth,
                      child: Center(
                        child: Container(
                          width: 50, // Fixed width for the pill shape
                          height: double.infinity,
                          decoration: BoxDecoration(
                            color: Colors.white.withOpacity(0.2),
                            borderRadius: BorderRadius.circular(20),
                            boxShadow: [
                              BoxShadow(
                                color: Colors.white.withOpacity(0.1),
                                blurRadius: 10,
                                spreadRadius: 1,
                              )
                            ],
                          ),
                        ),
                      ),
                    ),

                    // -------------------------------------------
                    // LAYER 2: The Icons (Foreground)
                    // -------------------------------------------
                    Row(
                      children: List.generate(icons.length, (index) {
                        return SizedBox(
                          width: itemWidth, // Force width to match calculations
                          child: GestureDetector(
                            behavior: HitTestBehavior.opaque,
                            onTap: () => onTap(index),
                            child: Center(
                              child: AnimatedScale(
                                duration: const Duration(milliseconds: 200),
                                scale: currentIndex == index ? 1.2 : 1.0,
                                child: Icon(
                                  icons[index],
                                  size: 28,
                                  color: currentIndex == index
                                      ? Colors.white
                                      : Colors.white.withOpacity(0.5),
                                ),
                              ),
                            ),
                          ),
                        );
                      }),
                    ),
                  ],
                );
              },
            ),
          ),
        ),
      ),
    );
  }
}
