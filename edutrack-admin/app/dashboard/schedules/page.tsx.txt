"use client"

import { useState, useEffect, useCallback } from "react"
import {
  Home,
  CalendarDays,
  Clock,
  Search,
  Loader2,
  Eye,
  X,
  Plus,
  Pencil,
  Trash2,
  BookOpen,
  GraduationCap,
} from "lucide-react"
import Link from "next/link"
import { usePathname } from "next/navigation"
import React from "react"
import {
  Breadcrumb,
  BreadcrumbItem,
  BreadcrumbLink,
  BreadcrumbList,
  BreadcrumbPage,
  BreadcrumbSeparator,
} from "@/components/ui/breadcrumb"
import {
  getSchedulesList,
  getScheduleStats,
  getClassesList,
  getTeachersList,
  getSubjectsList,
  createSchedule,
  updateSchedule,
  deleteSchedule,
} from "@/actions/dashboard"

// ── Types ──
interface KelasRef {
  id: number
  nama: string
}
interface MapelRef {
  id: number
  kode: string
  nama: string
}
interface TeacherRef {
  id: string
  name: string
}
interface ScheduleRow {
  id: number
  kelas_id: number
  mata_pelajaran_id: number
  teacher_id: string | null
  hari: string
  jam_mulai: string
  jam_selesai: string
  ruangan: string | null
  created_at: string
  kelas: KelasRef | null
  mata_pelajaran: MapelRef | null
  teachers: TeacherRef | null
}
interface ScheduleStatsData {
  total: number
  perDay: Record<string, number>
}

// ── Constants ──
const HARI_LIST = ["Senin", "Selasa", "Rabu", "Kamis", "Jumat", "Sabtu"]

const HARI_COLORS: Record<string, string> = {
  Senin: "bg-emerald-100 dark:bg-emerald-900/30 text-emerald-700 dark:text-emerald-400",
  Selasa: "bg-blue-100 dark:bg-blue-900/30 text-blue-700 dark:text-blue-400",
  Rabu: "bg-amber-100 dark:bg-amber-900/30 text-amber-700 dark:text-amber-400",
  Kamis: "bg-purple-100 dark:bg-purple-900/30 text-purple-700 dark:text-purple-400",
  Jumat: "bg-red-100 dark:bg-red-900/30 text-red-700 dark:text-red-400",
  Sabtu: "bg-cyan-100 dark:bg-cyan-900/30 text-cyan-700 dark:text-cyan-400",
}

const ROUTE_LABELS: Record<string, string> = {
  dashboard: "Overview",
  schedules: "Schedules",
}

function getLabel(segment: string): string {
  return ROUTE_LABELS[segment] ?? segment.charAt(0).toUpperCase() + segment.slice(1)
}

const EMPTY_FORM = {
  kelas_id: 0,
  mata_pelajaran_id: 0,
  teacher_id: "",
  hari: "Senin",
  jam_mulai: "07:00",
  jam_selesai: "07:45",
  ruangan: "",
}

export default function SchedulesPage() {
  const pathname = usePathname()
  const segments = pathname.split("/").filter(Boolean)
  const afterDashboard = segments.slice(1)
  const breadcrumbItems = afterDashboard.length === 0
    ? [{ label: "Overview", href: "/dashboard", isCurrent: true }]
    : afterDashboard.map((seg, i) => ({
        label: getLabel(seg),
        href: "/" + segments.slice(0, i + 2).join("/"),
        isCurrent: i === afterDashboard.length - 1,
      }))

  // Data state
  const [records, setRecords] = useState<ScheduleRow[]>([])
  const [stats, setStats] = useState<ScheduleStatsData>({ total: 0, perDay: {} })
  const [loading, setLoading] = useState(true)

  // Filter state
  const [search, setSearch] = useState("")
  const [filterHari, setFilterHari] = useState("")
  const [filterKelas, setFilterKelas] = useState("")

  // Dropdown data
  const [classesList, setClassesList] = useState<KelasRef[]>([])
  const [teachersList, setTeachersList] = useState<TeacherRef[]>([])
  const [subjectsList, setSubjectsList] = useState<MapelRef[]>([])

  // Modals
  const [detailRecord, setDetailRecord] = useState<ScheduleRow | null>(null)
  const [formOpen, setFormOpen] = useState(false)
  const [editingRecord, setEditingRecord] = useState<ScheduleRow | null>(null)
  const [formData, setFormData] = useState({ ...EMPTY_FORM })
  const [formLoading, setFormLoading] = useState(false)
  const [deleteConfirm, setDeleteConfirm] = useState<ScheduleRow | null>(null)
  const [deleteLoading, setDeleteLoading] = useState(false)

  // Fetch dropdown data once
  useEffect(() => {
    Promise.all([getClassesList(), getTeachersList(), getSubjectsList()]).then(
      ([classes, teachers, subjects]) => {
        setClassesList(classes as KelasRef[])
        setTeachersList(teachers as TeacherRef[])
        setSubjectsList(subjects as MapelRef[])
      }
    )
  }, [])

  const fetchRecords = useCallback(async () => {
    setLoading(true)
    const kelasId = filterKelas ? parseInt(filterKelas) : undefined
    const result = await getSchedulesList({
      kelasId,
      hari: filterHari || undefined,
    })
    let data = (result.data || []) as ScheduleRow[]

    // Client-side search filter
    if (search) {
      const q = search.toLowerCase()
      data = data.filter(
        (r) =>
          r.mata_pelajaran?.nama?.toLowerCase().includes(q) ||
          r.mata_pelajaran?.kode?.toLowerCase().includes(q) ||
          r.kelas?.nama?.toLowerCase().includes(q) ||
          r.teachers?.name?.toLowerCase().includes(q) ||
          r.ruangan?.toLowerCase().includes(q)
      )
    }

    setRecords(data)
    setLoading(false)
  }, [search, filterHari, filterKelas])

  const fetchStats = useCallback(async () => {
    const s = await getScheduleStats()
    setStats(s)
  }, [])

  useEffect(() => {
    fetchRecords()
  }, [fetchRecords])

  useEffect(() => {
    fetchStats()
  }, [fetchStats])

  // Group records by day for the grid view
  const groupedByDay: Record<string, ScheduleRow[]> = {}
  for (const day of HARI_LIST) {
    groupedByDay[day] = records.filter((r) => r.hari === day)
  }

  // ── CRUD handlers ──
  function openCreate() {
    setEditingRecord(null)
    setFormData({ ...EMPTY_FORM })
    setFormOpen(true)
  }

  function openEdit(record: ScheduleRow) {
    setEditingRecord(record)
    setFormData({
      kelas_id: record.kelas_id,
      mata_pelajaran_id: record.mata_pelajaran_id,
      teacher_id: record.teacher_id || "",
      hari: record.hari,
      jam_mulai: record.jam_mulai?.slice(0, 5) || "07:00",
      jam_selesai: record.jam_selesai?.slice(0, 5) || "07:45",
      ruangan: record.ruangan || "",
    })
    setFormOpen(true)
  }

  async function handleFormSubmit(e: React.FormEvent) {
    e.preventDefault()
    if (!formData.kelas_id || !formData.mata_pelajaran_id) return
    setFormLoading(true)

    const payload = {
      kelas_id: formData.kelas_id,
      mata_pelajaran_id: formData.mata_pelajaran_id,
      teacher_id: formData.teacher_id || null,
      hari: formData.hari,
      jam_mulai: formData.jam_mulai,
      jam_selesai: formData.jam_selesai,
      ruangan: formData.ruangan || null,
    }

    if (editingRecord) {
      await updateSchedule(editingRecord.id, payload)
    } else {
      await createSchedule(payload)
    }

    setFormLoading(false)
    setFormOpen(false)
    fetchRecords()
    fetchStats()
  }

  async function handleDelete() {
    if (!deleteConfirm) return
    setDeleteLoading(true)
    await deleteSchedule(deleteConfirm.id)
    setDeleteLoading(false)
    setDeleteConfirm(null)
    fetchRecords()
    fetchStats()
  }

  function formatTime(t: string | null) {
    if (!t) return "-"
    return t.slice(0, 5)
  }

  // Active days count
  const activeDays = Object.values(stats.perDay).filter((v) => v > 0).length

  return (
    <div className="space-y-6 max-w-7xl mx-auto px-4 sm:px-6 py-6">
      {/* Breadcrumb — mobile only */}
      <div className="lg:hidden">
        <Breadcrumb>
          <BreadcrumbList>
            <BreadcrumbItem>
              <BreadcrumbLink asChild>
                <Link href="/dashboard"><Home className="w-4 h-4" /></Link>
              </BreadcrumbLink>
            </BreadcrumbItem>
            {breadcrumbItems.map((item) => (
              <React.Fragment key={item.href}>
                <BreadcrumbSeparator />
                <BreadcrumbItem>
                  {item.isCurrent ? (
                    <BreadcrumbPage>{item.label}</BreadcrumbPage>
                  ) : (
                    <BreadcrumbLink asChild>
                      <Link href={item.href}>{item.label}</Link>
                    </BreadcrumbLink>
                  )}
                </BreadcrumbItem>
              </React.Fragment>
            ))}
          </BreadcrumbList>
        </Breadcrumb>
      </div>

      {/* Header */}
      <div className="flex flex-col sm:flex-row sm:items-center justify-between gap-4">
        <div>
          <h1 className="text-2xl md:text-3xl font-bold text-zinc-900 dark:text-zinc-50">Jadwal Pelajaran</h1>
          <p className="text-zinc-500 dark:text-zinc-400 mt-1">Kelola jadwal pelajaran sekolah</p>
        </div>
        <button
          onClick={openCreate}
          className="flex items-center gap-2 px-4 py-2.5 rounded-lg bg-emerald-600 hover:bg-emerald-700 text-white text-sm font-medium transition-colors self-start sm:self-auto"
        >
          <Plus className="w-4 h-4" />
          Tambah Jadwal
        </button>
      </div>

      {/* Stat Cards */}
      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
        <StatCard icon={CalendarDays} label="Total Jadwal" value={stats.total.toString()} />
        <StatCard icon={Clock} label="Hari Aktif" value={activeDays.toString()} color="emerald" />
        <StatCard icon={BookOpen} label="Mata Pelajaran" value={subjectsList.length.toString()} color="blue" />
        <StatCard icon={GraduationCap} label="Kelas" value={classesList.length.toString()} color="purple" />
      </div>

      {/* Search & Filter */}
      <div className="bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl p-4 md:p-6">
        <div className="flex flex-col sm:flex-row gap-3">
          <div className="relative flex-1">
            <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-zinc-400" />
            <input
              type="text"
              value={search}
              onChange={(e) => setSearch(e.target.value)}
              placeholder="Cari mata pelajaran, guru, kelas, ruangan..."
              className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg pl-10 pr-4 py-2 text-sm text-zinc-900 dark:text-zinc-50 placeholder:text-zinc-500 focus:outline-none focus:ring-2 focus:ring-zinc-900 dark:focus:ring-zinc-100 transition-shadow"
            />
          </div>
          <select
            value={filterHari}
            onChange={(e) => setFilterHari(e.target.value)}
            className="bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-zinc-900 dark:focus:ring-zinc-100 transition-shadow sm:w-40"
          >
            <option value="">Semua Hari</option>
            {HARI_LIST.map((h) => (
              <option key={h} value={h}>{h}</option>
            ))}
          </select>
          <select
            value={filterKelas}
            onChange={(e) => setFilterKelas(e.target.value)}
            className="bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-zinc-900 dark:focus:ring-zinc-100 transition-shadow sm:w-44"
          >
            <option value="">Semua Kelas</option>
            {classesList.map((k) => (
              <option key={k.id} value={k.id}>{k.nama}</option>
            ))}
          </select>
        </div>
      </div>

      {/* Schedule Grid — Day-based view */}
      {!filterHari ? (
        <div className="space-y-4">
          {HARI_LIST.map((day) => {
            const dayRecords = groupedByDay[day] || []
            if (dayRecords.length === 0) return null
            return (
              <div key={day} className="bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl overflow-hidden">
                <div className="px-4 sm:px-6 py-3 border-b border-zinc-200 dark:border-zinc-800 bg-zinc-50/50 dark:bg-zinc-900/50 flex items-center justify-between">
                  <div className="flex items-center gap-3">
                    <span className={`inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium ${HARI_COLORS[day] || "bg-zinc-100 dark:bg-zinc-800 text-zinc-600 dark:text-zinc-400"}`}>
                      {day}
                    </span>
                    <span className="text-sm text-zinc-500 dark:text-zinc-400">
                      {dayRecords.length} jadwal
                    </span>
                  </div>
                </div>
                <div className="overflow-x-auto">
                  <ScheduleTable
                    records={dayRecords}
                    onDetail={setDetailRecord}
                    onEdit={openEdit}
                    onDelete={setDeleteConfirm}
                    showDay={false}
                  />
                </div>
              </div>
            )
          })}
          {records.length === 0 && !loading && (
            <div className="bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl p-12 text-center text-zinc-500">
              Tidak ada jadwal ditemukan.
            </div>
          )}
        </div>
      ) : (
        /* Flat table when day filter is active */
        <div className="bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl overflow-hidden">
          <div className="overflow-x-auto">
            <ScheduleTable
              records={records}
              onDetail={setDetailRecord}
              onEdit={openEdit}
              onDelete={setDeleteConfirm}
              showDay={true}
            />
          </div>
          {records.length === 0 && !loading && (
            <div className="p-12 text-center text-zinc-500">
              Tidak ada jadwal ditemukan.
            </div>
          )}
        </div>
      )}

      {/* Loading state */}
      {loading && (
        <div className="flex items-center justify-center py-12 text-zinc-500">
          <Loader2 className="w-5 h-5 animate-spin mr-2" />
          Loading...
        </div>
      )}

      {/* Detail Modal */}
      {detailRecord && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 p-4">
          <div className="w-full max-w-md bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl p-6 shadow-2xl">
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-lg font-bold text-zinc-900 dark:text-zinc-50">Detail Jadwal</h2>
              <button
                onClick={() => setDetailRecord(null)}
                className="p-1 text-zinc-400 hover:text-zinc-900 dark:hover:text-zinc-50 rounded-md hover:bg-zinc-100 dark:hover:bg-zinc-800 transition-colors"
              >
                <X className="w-5 h-5" />
              </button>
            </div>

            <div className="space-y-4">
              <DetailRow label="Hari" value={detailRecord.hari} />
              <DetailRow label="Kelas" value={detailRecord.kelas?.nama || "-"} />
              <DetailRow label="Mata Pelajaran" value={detailRecord.mata_pelajaran ? `${detailRecord.mata_pelajaran.nama} (${detailRecord.mata_pelajaran.kode})` : "-"} />
              <DetailRow label="Guru" value={detailRecord.teachers?.name || "-"} />
              <DetailRow label="Jam" value={`${formatTime(detailRecord.jam_mulai)} - ${formatTime(detailRecord.jam_selesai)}`} />
              <DetailRow label="Ruangan" value={detailRecord.ruangan || "-"} />
            </div>

            <div className="flex justify-between mt-6">
              <div className="flex gap-2">
                <button
                  onClick={() => { setDetailRecord(null); openEdit(detailRecord) }}
                  className="flex items-center gap-1.5 px-3 py-2 rounded-lg text-sm font-medium text-emerald-700 dark:text-emerald-400 border border-emerald-200 dark:border-emerald-800 hover:bg-emerald-50 dark:hover:bg-emerald-900/20 transition-colors"
                >
                  <Pencil className="w-3.5 h-3.5" />
                  Edit
                </button>
                <button
                  onClick={() => { setDetailRecord(null); setDeleteConfirm(detailRecord) }}
                  className="flex items-center gap-1.5 px-3 py-2 rounded-lg text-sm font-medium text-red-700 dark:text-red-400 border border-red-200 dark:border-red-800 hover:bg-red-50 dark:hover:bg-red-900/20 transition-colors"
                >
                  <Trash2 className="w-3.5 h-3.5" />
                  Hapus
                </button>
              </div>
              <button
                onClick={() => setDetailRecord(null)}
                className="px-4 py-2 rounded-lg text-sm font-medium text-zinc-700 dark:text-zinc-300 border border-zinc-200 dark:border-zinc-800 hover:bg-zinc-50 dark:hover:bg-zinc-900 transition-colors"
              >
                Tutup
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Create/Edit Modal */}
      {formOpen && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 p-4">
          <div className="w-full max-w-lg bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl p-6 shadow-2xl max-h-[90vh] overflow-y-auto">
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-lg font-bold text-zinc-900 dark:text-zinc-50">
                {editingRecord ? "Edit Jadwal" : "Tambah Jadwal"}
              </h2>
              <button
                onClick={() => setFormOpen(false)}
                className="p-1 text-zinc-400 hover:text-zinc-900 dark:hover:text-zinc-50 rounded-md hover:bg-zinc-100 dark:hover:bg-zinc-800 transition-colors"
              >
                <X className="w-5 h-5" />
              </button>
            </div>

            <form onSubmit={handleFormSubmit} className="space-y-4">
              {/* Hari */}
              <div>
                <label className="block text-sm font-medium text-zinc-700 dark:text-zinc-300 mb-1">Hari</label>
                <select
                  value={formData.hari}
                  onChange={(e) => setFormData({ ...formData, hari: e.target.value })}
                  className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-emerald-500"
                >
                  {HARI_LIST.map((h) => (
                    <option key={h} value={h}>{h}</option>
                  ))}
                </select>
              </div>

              {/* Kelas */}
              <div>
                <label className="block text-sm font-medium text-zinc-700 dark:text-zinc-300 mb-1">Kelas <span className="text-red-500">*</span></label>
                <select
                  value={formData.kelas_id}
                  onChange={(e) => setFormData({ ...formData, kelas_id: parseInt(e.target.value) || 0 })}
                  required
                  className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-emerald-500"
                >
                  <option value={0}>Pilih Kelas</option>
                  {classesList.map((k) => (
                    <option key={k.id} value={k.id}>{k.nama}</option>
                  ))}
                </select>
              </div>

              {/* Mata Pelajaran */}
              <div>
                <label className="block text-sm font-medium text-zinc-700 dark:text-zinc-300 mb-1">Mata Pelajaran <span className="text-red-500">*</span></label>
                <select
                  value={formData.mata_pelajaran_id}
                  onChange={(e) => setFormData({ ...formData, mata_pelajaran_id: parseInt(e.target.value) || 0 })}
                  required
                  className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-emerald-500"
                >
                  <option value={0}>Pilih Mata Pelajaran</option>
                  {subjectsList.map((s) => (
                    <option key={s.id} value={s.id}>{s.nama} ({s.kode})</option>
                  ))}
                </select>
              </div>

              {/* Guru */}
              <div>
                <label className="block text-sm font-medium text-zinc-700 dark:text-zinc-300 mb-1">Guru</label>
                <select
                  value={formData.teacher_id}
                  onChange={(e) => setFormData({ ...formData, teacher_id: e.target.value })}
                  className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-emerald-500"
                >
                  <option value="">Belum ditentukan</option>
                  {teachersList.map((t) => (
                    <option key={t.id} value={t.id}>{t.name}</option>
                  ))}
                </select>
              </div>

              {/* Jam Mulai & Jam Selesai */}
              <div className="grid grid-cols-2 gap-3">
                <div>
                  <label className="block text-sm font-medium text-zinc-700 dark:text-zinc-300 mb-1">Jam Mulai</label>
                  <input
                    type="time"
                    value={formData.jam_mulai}
                    onChange={(e) => setFormData({ ...formData, jam_mulai: e.target.value })}
                    required
                    className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-emerald-500"
                  />
                </div>
                <div>
                  <label className="block text-sm font-medium text-zinc-700 dark:text-zinc-300 mb-1">Jam Selesai</label>
                  <input
                    type="time"
                    value={formData.jam_selesai}
                    onChange={(e) => setFormData({ ...formData, jam_selesai: e.target.value })}
                    required
                    className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 focus:outline-none focus:ring-2 focus:ring-emerald-500"
                  />
                </div>
              </div>

              {/* Ruangan */}
              <div>
                <label className="block text-sm font-medium text-zinc-700 dark:text-zinc-300 mb-1">Ruangan</label>
                <input
                  type="text"
                  value={formData.ruangan}
                  onChange={(e) => setFormData({ ...formData, ruangan: e.target.value })}
                  placeholder="Contoh: Ruang 101"
                  className="w-full bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg px-3 py-2 text-sm text-zinc-900 dark:text-zinc-50 placeholder:text-zinc-500 focus:outline-none focus:ring-2 focus:ring-emerald-500"
                />
              </div>

              {/* Actions */}
              <div className="flex justify-end gap-3 pt-2">
                <button
                  type="button"
                  onClick={() => setFormOpen(false)}
                  className="px-4 py-2 rounded-lg text-sm font-medium text-zinc-700 dark:text-zinc-300 border border-zinc-200 dark:border-zinc-800 hover:bg-zinc-50 dark:hover:bg-zinc-900 transition-colors"
                >
                  Batal
                </button>
                <button
                  type="submit"
                  disabled={formLoading || !formData.kelas_id || !formData.mata_pelajaran_id}
                  className="flex items-center gap-2 px-4 py-2 rounded-lg text-sm font-medium bg-emerald-600 hover:bg-emerald-700 text-white transition-colors disabled:opacity-50 disabled:cursor-not-allowed"
                >
                  {formLoading && <Loader2 className="w-4 h-4 animate-spin" />}
                  {editingRecord ? "Simpan Perubahan" : "Tambah Jadwal"}
                </button>
              </div>
            </form>
          </div>
        </div>
      )}

      {/* Delete Confirmation Modal */}
      {deleteConfirm && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 p-4">
          <div className="w-full max-w-sm bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl p-6 shadow-2xl">
            <div className="flex items-center justify-between mb-4">
              <h2 className="text-lg font-bold text-zinc-900 dark:text-zinc-50">Hapus Jadwal</h2>
              <button
                onClick={() => setDeleteConfirm(null)}
                className="p-1 text-zinc-400 hover:text-zinc-900 dark:hover:text-zinc-50 rounded-md hover:bg-zinc-100 dark:hover:bg-zinc-800 transition-colors"
              >
                <X className="w-5 h-5" />
              </button>
            </div>
            <p className="text-sm text-zinc-600 dark:text-zinc-400 mb-2">
              Apakah Anda yakin ingin menghapus jadwal ini?
            </p>
            <div className="bg-zinc-50 dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg p-3 mb-6 text-sm space-y-1">
              <p className="font-medium text-zinc-900 dark:text-zinc-50">{deleteConfirm.mata_pelajaran?.nama || "-"}</p>
              <p className="text-zinc-500">{deleteConfirm.hari} &middot; {formatTime(deleteConfirm.jam_mulai)} - {formatTime(deleteConfirm.jam_selesai)}</p>
              <p className="text-zinc-500">{deleteConfirm.kelas?.nama || "-"}</p>
            </div>
            <div className="flex justify-end gap-3">
              <button
                onClick={() => setDeleteConfirm(null)}
                className="px-4 py-2 rounded-lg text-sm font-medium text-zinc-700 dark:text-zinc-300 border border-zinc-200 dark:border-zinc-800 hover:bg-zinc-50 dark:hover:bg-zinc-900 transition-colors"
              >
                Batal
              </button>
              <button
                onClick={handleDelete}
                disabled={deleteLoading}
                className="flex items-center gap-2 px-4 py-2 rounded-lg text-sm font-medium bg-red-600 hover:bg-red-700 text-white transition-colors disabled:opacity-50"
              >
                {deleteLoading && <Loader2 className="w-4 h-4 animate-spin" />}
                Hapus
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  )
}

function DetailRow({ label, value }: { label: string; value: string }) {

      <p className="text-sm font-medium text-zinc-900 dark:text-zinc-50 mt-0.5">{value}</p>

}

function StatCard({ icon: Icon, label, value, color }: { icon: React.ElementType; label: string; value: string; color?: string }) {
  const iconColorMap: Record<string, string> = {
    emerald: "text-emerald-600 dark:text-emerald-400",
    amber: "text-amber-600 dark:text-amber-400",
    blue: "text-blue-600 dark:text-blue-400",
    red: "text-red-600 dark:text-red-400",
    purple: "text-purple-600 dark:text-purple-400",
  }

  const bgColorMap: Record<string, string> = {
    emerald: "bg-emerald-100 dark:bg-emerald-900/30 border-emerald-200 dark:border-emerald-800",
    amber: "bg-amber-100 dark:bg-amber-900/30 border-amber-200 dark:border-amber-800",
    blue: "bg-blue-100 dark:bg-blue-900/30 border-blue-200 dark:border-blue-800",
    red: "bg-red-100 dark:bg-red-900/30 border-red-200 dark:border-red-800",
    purple: "bg-purple-100 dark:bg-purple-900/30 border-purple-200 dark:border-purple-800",
  }

  return (
    <div className="bg-white dark:bg-zinc-950 border border-zinc-200 dark:border-zinc-800 rounded-2xl p-5 relative overflow-hidden flex flex-col justify-between min-h-[120px]">
      <div className="flex items-center gap-3 mb-3">
        <div className={`w-8 h-8 rounded-lg flex items-center justify-center border ${color ? bgColorMap[color] || "bg-zinc-100 dark:bg-zinc-900 border-zinc-200 dark:border-zinc-800" : "bg-zinc-100 dark:bg-zinc-900 border-zinc-200 dark:border-zinc-800"}`}>
          <Icon className={`w-4 h-4 ${color ? iconColorMap[color] || "text-zinc-600 dark:text-zinc-400" : "text-zinc-600 dark:text-zinc-400"}`} />
        </div>
        <span className="text-sm font-medium text-zinc-600 dark:text-zinc-400">{label}</span>
      </div>
      <div className="text-3xl font-bold text-zinc-900 dark:text-zinc-50">{value}</div>
      <Icon className="absolute -bottom-4 -right-4 w-24 h-24 text-zinc-50 dark:text-zinc-900/50 opacity-50 z-0" strokeWidth={1} />
    </div>
  )
}
